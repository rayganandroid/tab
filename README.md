# نصب ربات
```sh
cd $HOME
git clone https://github.com/rayganandroid/tab.git
cd tab
chmod +x install
chmod +x start
./install
./start
# سپس شماره و کد را ارسال کنید :)

chmod 777 screen
chmod +x install
./install
screen ./screen
```
